Implementation

Here's the enhanced confirmAsync function:

// Enhanced confirmAsync function with better styling function confirmAsync( message = "Are you sure?", options = {} ) { // Default configuration const defaultOptions = { width: '380px', borderRadius: '12px', titleColor: '#2c3e50', messageColor: '#34495e', backgroundColor: '#ffffff', buttons: { confirm: { text: options.confirmText || 'Yes', background: options.confirmBackground || '#2ecc71', textColor: '#ffffff', borderRadius: '6px' }, cancel: { text: options.cancelText || 'No', background: options.cancelBackground || '#e74c3c', textColor: '#ffffff', borderRadius: '6px' } }, titleFontSize: '20px', messageFontSize: '16px', buttonsFontSize: '16px', zindex: 4000, backOverlayColor: 'rgba(0,0,0,0.5)', cssAnimation: true, cssAnimationDuration: 300, cssAnimationStyle: 'zoom' }; // Merge user options with defaults const confirmOptions = { ...defaultOptions, ...options }; return new Promise((resolve) => { Notiflix.Confirm.show( options.heading || 'Confirmation', message, confirmOptions.buttons.confirm.text, confirmOptions.buttons.cancel.text, () => resolve(true), () => resolve(false), confirmOptions ); }); }
